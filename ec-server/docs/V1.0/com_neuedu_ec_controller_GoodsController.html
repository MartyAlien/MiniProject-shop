商品列表搜索

请求URL

/goods/search GET POST

请求参数 application/x-www-form-urlencoded

参数名 类型 必须 描述
query string 否 关键字
cid int 否 分类
pagenum int 是 页码
pagesize int 是 页容量

返回结果

{
	"meta":{
		"code":"int",
		"msg":"string"
	},
	"message":{
		"total":"int",
		"pagenum":"int",
		"goods":[{
			"goods_id":"int //商品ID",
			"cat_id":"int //分类id",
			"goods_name":"string //商品名称",
			"goods_price":"double //商品价格",
			"goods_number":"int //商品数量",
			"goods_weight":"int //商品重量",
			"goods_big_logo":"string //商品大图标",
			"goods_small_logo":"string //商品小图标",
			"goods_introduce":"string //商品介绍",
			"add_time":"long //商品添加时间",
			"upd_time":"long //商品更新时间",
			"hot_mumber":"int //热门商品数",
			"cat_one_id":"int //所属一级分类",
			"cat_two_id":"int //所属二级分类",
			"cat_three_id":"int //所属三级分类",
			"is_del":"string //是否删除",
			"is_promote":"boolean //是否促销",
			"goods_state":"int //商品状态",
			"goods_cat":"string //商品分类",
			"delete_time":"date //删除时间",
			"createtime":"date //数据创建时间",
			"pics":[{
				"pics_id":"int //主键",
				"goods_id":"int //商品ID",
				"pics_big":"string //大图",
				"pics_mid":"string //中图",
				"pics_sma":"string //小图",
				"pics_big_url":"string //大图路径",
				"pics_mid_url":"string //中图路径",
				"pics_sma_url":"string //小图路径",
				"createtime":"date //创建时间"
			}],
			"attrs":[{
				"attr_id":"int //主键",
				"goods_id":"int //商品ID",
				"attr_name":"string //属性名称",
				"attr_value":"string //属性属性值",
				"add_price":"int //属性价格",
				"atrr_vals":"string //属性多值",
				"attr_sel":"string //attr_sel",
				"attr_write":"string //attr_write",
				"createtime":"date //创建时间"
			}]
		}]
	}
}
